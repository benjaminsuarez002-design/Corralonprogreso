function getOnePaceAccessToken(request, url) {
  const queryToken = url.searchParams.get('access_token') || url.searchParams.get('token') || '';
  if (queryToken) return queryToken.trim();

  const auth = request.headers.get('authorization') || request.headers.get('Authorization') || '';
  if (auth.toLowerCase().startsWith('bearer ')) return auth.slice(7).trim();
  return '';
}

function cleanDriveFileId(value) {
  const fileId = String(value || '').trim();
  return /^[A-Za-z0-9_-]{10,}$/.test(fileId) ? fileId : '';
}

async function handleOnePaceVideo(request) {
  const url = new URL(request.url);

  if (request.method === 'OPTIONS') {
    return new Response(null, { status: 204, headers: videoHeaders() });
  }

  if (request.method !== 'GET' && request.method !== 'HEAD') {
    return videoError('Method not allowed', 405);
  }

  const fileId = cleanDriveFileId(url.searchParams.get('fileId') || url.searchParams.get('id'));
  if (!fileId) return videoError('Falta fileId', 400);

  let accessToken = getOnePaceAccessToken(request, url);
  if (!accessToken) accessToken = await getDriveServiceToken();

  const driveHeaders = {
    Authorization: `Bearer ${accessToken}`
  };
  const range = request.headers.get('range') || request.headers.get('Range');
  if (range) driveHeaders.Range = range;

  const driveResponse = await fetch(
    `https://www.googleapis.com/drive/v3/files/${encodeURIComponent(fileId)}?alt=media`,
    {
      method: 'GET',
      headers: driveHeaders,
      cf: {
        cacheTtl: 0,
        cacheEverything: false
      }
    }
  );

  const outputHeaders = new Headers();
  [
    'content-type',
    'content-length',
    'content-range',
    'accept-ranges',
    'etag',
    'last-modified'
  ].forEach(name => {
    const value = driveResponse.headers.get(name);
    if (value) outputHeaders.set(name, value);
  });

  if (!outputHeaders.has('content-type')) {
    outputHeaders.set('content-type', url.searchParams.get('mimeType') || 'video/mp4');
  }
  outputHeaders.set('accept-ranges', 'bytes');
  outputHeaders.set('cache-control', 'no-store');
  Object.entries(VIDEO_CORS_HEADERS).forEach(([key, value]) => outputHeaders.set(key, value));

  if (!driveResponse.ok) {
    const message = await driveResponse.text().catch(() => '');
    return new Response(message || `Drive respondio ${driveResponse.status}`, {
      status: driveResponse.status,
      statusText: driveResponse.statusText,
      headers: outputHeaders
    });
  }

  return new Response(request.method === 'HEAD' ? null : driveResponse.body, {
    status: driveResponse.status,
    statusText: driveResponse.statusText,
    headers: outputHeaders
  });
}

async function handleOnePaceCatalog(request) {
  const url = new URL(request.url);

  if (request.method === 'OPTIONS') {
    return new Response(null, { status: 204, headers: JSON_HEADERS });
  }

  if (request.method !== 'GET') {
    return json({ ok: false, error: 'Method not allowed' }, 405);
  }

  const folderId = cleanDriveFileId(url.searchParams.get('folderId')) || ONEPACE_DEFAULT_FOLDER_ID;
  const token = await getDriveServiceToken();
  const files = [];
  let pageToken = '';

  do {
    const driveUrl = new URL('https://www.googleapis.com/drive/v3/files');
    driveUrl.searchParams.set('q', `'${folderId}' in parents and trashed=false`);
    driveUrl.searchParams.set('fields', 'nextPageToken,files(id,name,mimeType,size,modifiedTime,webViewLink)');
    driveUrl.searchParams.set('pageSize', '1000');
    driveUrl.searchParams.set('orderBy', 'name_natural');
    if (pageToken) driveUrl.searchParams.set('pageToken', pageToken);

    const res = await fetch(driveUrl.toString(), {
      headers: { Authorization: `Bearer ${token}` }
    });
    const data = await res.json().catch(() => ({}));
    if (!res.ok) {
      return json({ ok: false, error: data.error?.message || `Drive catalogo respondio ${res.status}` }, res.status);
    }

    files.push(...(data.files || []));
    pageToken = data.nextPageToken || '';
  } while (pageToken);

  const videos = files.filter(file => {
    const name = String(file.name || '');
    return String(file.mimeType || '').startsWith('video/') || /\.(mp4|mkv|webm|mov|m4v)$/i.test(name);
  });

  return json({ ok: true, files: videos, updatedAt: Date.now() });
}

function toBase64(bytes) {
  let binary = '';
  bytes.forEach(b => { binary += String.fromCharCode(b); });
  return btoa(binary);
}

function fromBase64(base64) {
  const binary = atob(base64);
  const out = new Uint8Array(binary.length);
  for (let i = 0; i < binary.length; i++) out[i] = binary.charCodeAt(i);
  return out;
}

function generateTokenHex(length = 32) {

function escapeHtml(value) {
  return String(value || '')
    .replaceAll('&', '&amp;')
    .replaceAll('<', '&lt;')
    .replaceAll('>', '&gt;')
    .replaceAll('"', '&quot;')
    .replaceAll("'", '&#39;');
}

async function getAesKey(secret) {
  const digest = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(secret));
  return crypto.subtle.importKey('raw', digest, { name: 'AES-GCM' }, false, ['encrypt', 'decrypt']);
}

async function encryptPassword(password) {
  const key = await getAesKey(PASSWORD_ENCRYPTION_SECRET);
  const iv = crypto.getRandomValues(new Uint8Array(12));
  const ciphertext = await crypto.subtle.encrypt(
    { name: 'AES-GCM', iv },
    key,
    new TextEncoder().encode(password)
  );
  return {
    password_ciphertext: toBase64(new Uint8Array(ciphertext)),
    password_iv: toBase64(iv),
    password_version: 1
  };
}

async function decryptPassword(ciphertextBase64, ivBase64) {
  const key = await getAesKey(PASSWORD_ENCRYPTION_SECRET);
  const plaintext = await crypto.subtle.decrypt(
    { name: 'AES-GCM', iv: fromBase64(ivBase64) },
    key,
    fromBase64(ciphertextBase64)
  );
  return new TextDecoder().decode(plaintext);
}

  };
}

async function requireAdmin(request, action = 'admin_access') {
  const idToken = getBearerToken(request);
  if (!idToken) {
    return { ok: false, status: 401, error: 'Falta token de sesión' };
  }

  const user = await verifyFirebaseIdToken(idToken);
  if (!user?.email) {
    await insertAdminAudit({
      admin_email: 'desconocido',
      action,
      status: 'denied',
      detail: 'Token inválido o vencido'
    });
    return { ok: false, status: 401, error: 'Sesión inválida o vencida' };
  }

  if (user.email !== ADMIN_EMAIL) {
    await insertAdminAudit({
      admin_email: user.email,
      target_uid: user.uid || null,
      action,
      status: 'denied',
      detail: 'Email sin permisos de admin'
    });
    return { ok: false, status: 403, error: 'No tenés permisos para acceder al panel admin' };
  }

  const pin = String(request.headers.get('x-admin-pin') || '').trim();
  if (!pin || pin !== ADMIN_PANEL_PIN) {
    await insertAdminAudit({
      admin_email: user.email,
      target_uid: user.uid || null,
      action,
      status: 'denied',
      detail: 'PIN admin inválido'
    });
    return { ok: false, status: 403, error: 'PIN admin incorrecto' };
  }

  return { ok: true, adminEmail: user.email, adminUid: user.uid, idToken };
}

async function confirmarReset(token, email, codigo, nuevaPass) {
  const resAuth = await fetch(
    `https://identitytoolkit.googleapis.com/v1/projects/${FIREBASE_PROJECT_ID}/accounts:lookup`,
    {
      method: 'POST',
      headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({ email: [email] }),
    }
  );
  const dataAuth = await resAuth.json();
  const uid = dataAuth.users?.[0]?.localId;
  if (!uid) return { ok: false, error: 'Usuario no encontrado' };

  const urlDoc = `https://firestore.googleapis.com/v1/projects/${FIREBASE_PROJECT_ID}/databases/(default)/documents/resetCodigos/${uid}`;
  const resDoc = await fetch(urlDoc, { headers: { Authorization: `Bearer ${token}` } });
  const doc = await resDoc.json();
  if (!doc.fields) return { ok: false, error: 'Código no encontrado' };

  const codigoGuardado = doc.fields.codigo?.stringValue;
  const expira = parseInt(doc.fields.expira?.integerValue || '0', 10);
  const usado = doc.fields.usado?.booleanValue;

  if (usado) return { ok: false, error: 'Este código ya fue usado' };
  if (Date.now() > expira) return { ok: false, error: 'El código expiró. Pedí uno nuevo' };
  if (codigoGuardado !== codigo) return { ok: false, error: 'Código incorrecto' };

  const resPass = await fetch(
    `https://identitytoolkit.googleapis.com/v1/projects/${FIREBASE_PROJECT_ID}/accounts:update`,
    {
      method: 'POST',
      headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({ localId: uid, password: nuevaPass }),
    }
  );
  const dataPass = await resPass.json();
  if (dataPass.error) return { ok: false, error: 'Error al cambiar la contraseña' };

  await fetch(urlDoc, {
    method: 'PATCH',
    headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
    body: JSON.stringify({ fields: { ...doc.fields, usado: { booleanValue: true } } }),
  });

  return { ok: true };
}

async function crearRegistroPendiente(body) {
  const nombre = String(body.nombre || '').trim();
  const email = normalizeEmail(body.email);
  const password = String(body.password || '');
  const comprobanteUrl = String(body.comprobanteUrl || '').trim();

  if (!nombre || !email || !password || !comprobanteUrl) {
    return { ok: false, error: 'Faltan datos obligatorios', status: 400 };
  }
  if (!validateEmail(email)) {
    return { ok: false, error: 'Email inválido', status: 400 };
  }
  if (password.length < 6) {
    return { ok: false, error: 'La contraseña debe tener al menos 6 caracteres', status: 400 };
  }

  const pending = await findPendingByEmail(email);
  if (pending) {
    return { ok: false, error: 'Ya existe una solicitud pendiente para este email', status: 409 };
  }

  const exists = await isFirebaseEmailRegistered(email);
  if (exists) {
    return { ok: false, error: 'Ya existe una cuenta creada con ese email', status: 409 };
  }

  const encrypted = await encryptPassword(password);
  const approvalToken = generateTokenHex(32);
  const row = await insertPendingRecord({
    nombre,
    email,
    ...encrypted,
    comprobante_url: comprobanteUrl,
    estado: 'pendiente',
    approval_token: approvalToken,
    approval_token_expires_at: new Date(Date.now() + TOKEN_TTL_MS).toISOString()
  });

  await sendApprovalEmail(row, approvalToken);
  return { ok: true, estado: 'pendiente', id: row.id, status: 200 };
}

async function obtenerRegistroPendiente(tokenValue) {
  if (!tokenValue) return { ok: false, error: 'Falta token', status: 400 };
  let row = await findPendingByToken(tokenValue);
  if (!row) return { ok: false, error: 'Token inválido o inexistente', status: 404 };
  row = await markExpiredIfNeeded(row);
  return { ...mapPendingRow(row), status: 200 };
}

async function aprobarRegistroPendiente(tokenValue) {
  if (!tokenValue) return { ok: false, error: 'Falta token', status: 400 };
  let row = await findPendingByToken(tokenValue);
  if (!row) return { ok: false, error: 'Token inválido o inexistente', status: 404 };
  row = await markExpiredIfNeeded(row);
  if (row.estado !== 'pendiente') {
    return { ok: false, error: `La solicitud ya está ${row.estado}`, ...mapPendingRow(row), status: 409 };
  }

  const exists = await isFirebaseEmailRegistered(row.email);
  if (exists) {
    const updated = await updatePendingRecord(row.id, {
      estado: 'rechazado',
      rejected_at: new Date().toISOString()
    });
    return { ok: false, error: 'Ya existe una cuenta creada con ese email', ...mapPendingRow(updated), status: 409 };
  }

  const password = await decryptPassword(row.password_ciphertext, row.password_iv);
  const firebaseUid = await createFirebaseUser(row.email, password);
  const updated = await updatePendingRecord(row.id, {
    estado: 'aprobado',
    approved_at: new Date().toISOString(),
    firebase_uid: firebaseUid
  });
  return { ...mapPendingRow(updated), status: 200 };
}

}

async function aprobarRegistroPendienteInterno({ tokenValue = '', idValue = '', adminEmail = '' } = {}) {
  const row = await cargarPendientePorReferencia({ tokenValue, idValue });
  if (!row) return { ok: false, error: 'Solicitud no encontrada', status: 404 };

  const result = tokenValue
    ? await aprobarRegistroPendiente(tokenValue)
    : await (async () => {
        let current = await findPendingById(idValue);
        if (!current) return { ok: false, error: 'Solicitud no encontrada', status: 404 };
        current = await markExpiredIfNeeded(current);
        if (current.estado !== 'pendiente') {
          return { ok: false, error: `La solicitud ya está ${current.estado}`, ...mapPendingRow(current), status: 409 };
        }
        const exists = await isFirebaseEmailRegistered(current.email);
        if (exists) {
          const updated = await updatePendingRecord(current.id, {
            estado: 'rechazado',
            rejected_at: new Date().toISOString()
          });
          return { ok: false, error: 'Ya existe una cuenta creada con ese email', ...mapPendingRow(updated), status: 409 };
        }
        const password = await decryptPassword(current.password_ciphertext, current.password_iv);
        const firebaseUid = await createFirebaseUser(current.email, password);
        const updated = await updatePendingRecord(current.id, {
          estado: 'aprobado',
          approved_at: new Date().toISOString(),
          firebase_uid: firebaseUid
        });
        return { ...mapPendingRow(updated), status: 200 };
      })();

  await insertAdminAudit({
    admin_email: adminEmail || ADMIN_EMAIL,
    action: 'approve_pending',
    target_email: row.email || '',
    target_uid: result.firebaseUid || row.firebase_uid || null,
    status: result.ok === false ? 'error' : 'success',
    detail: result.ok === false ? (result.error || 'Error al aprobar pendiente') : `Pendiente ${row.id} aprobado`
  });

  return result;
}

async function rechazarRegistroPendienteInterno({ tokenValue = '', idValue = '', adminEmail = '' } = {}) {
  const row = await cargarPendientePorReferencia({ tokenValue, idValue });
  if (!row) return { ok: false, error: 'Solicitud no encontrada', status: 404 };

  const result = tokenValue
    ? await rechazarRegistroPendiente(tokenValue)
    : await (async () => {
        let current = await findPendingById(idValue);
        if (!current) return { ok: false, error: 'Solicitud no encontrada', status: 404 };
        current = await markExpiredIfNeeded(current);
        if (current.estado !== 'pendiente') {
          return { ok: false, error: `La solicitud ya está ${current.estado}`, ...mapPendingRow(current), status: 409 };
        }
        const updated = await updatePendingRecord(current.id, {
          estado: 'rechazado',
          rejected_at: new Date().toISOString()
        });
        return { ...mapPendingRow(updated), status: 200 };
      })();

  await insertAdminAudit({
    admin_email: adminEmail || ADMIN_EMAIL,
    action: 'reject_pending',
    target_email: row.email || '',
    target_uid: row.firebase_uid || null,
    status: result.ok === false ? 'error' : 'success',
    detail: result.ok === false ? (result.error || 'Error al rechazar pendiente') : `Pendiente ${row.id} rechazado`
  });

  return result;
}

    const data = await adminProjectRequest(adminToken, `accounts:batchGet?${qs.toString()}`, { method: 'GET' });
    users = users.concat(data.users || []);
    nextPageToken = data.nextPageToken || '';
  } while (nextPageToken && users.length < 5000);

  return users.map(u => ({
    uid: u.localId || '',
    email: u.email || '',
    disabled: !!u.disabled,
    createdAt: u.createdAt || '',
    lastLoginAt: u.lastLoginAt || ''
  }));
}

async function lookupFirebaseUser(adminToken, uid) {
  const data = await adminProjectRequest(adminToken, 'accounts:lookup', {
    method: 'POST',
    body: { localId: [uid] }
  });
  return data.users?.[0] || null;
}

async function setFirebaseUserDisabled(adminToken, uid, disabled) {
  return adminProjectRequest(adminToken, 'accounts:update', {
    method: 'POST',
    body: { localId: uid, disableUser: !!disabled }
  });
}

async function deleteFirebaseUser(adminToken, uid) {
  return adminProjectRequest(adminToken, 'accounts:delete', {
    method: 'POST',
    body: { localId: uid }
  });
}

function htmlAttrSharePreview(value) {
  return String(value ?? '')
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

function cleanHttpUrlSharePreview(value) {
  const raw = String(value || '').trim();
  if (!raw) return '';
  try {
    const url = new URL(raw);
    return url.protocol === 'http:' || url.protocol === 'https:' ? url.toString() : '';
  } catch (_) {
    return '';
  }
}

function buildCorralonShareMeta(url) {
  const codigo = String(url.searchParams.get('articulo') || '').trim();
  const image = cleanHttpUrlSharePreview(url.searchParams.get('previewImg'));
  if (!codigo || !image) return '';

  const title = String(url.searchParams.get('previewTitulo') || 'Articulo Corralon Progreso').trim();
  const price = String(url.searchParams.get('previewPrecio') || '').trim();
  const description = price
    ? `${price} - Codigo: ${codigo} - Corralon Progreso`
    : `Codigo: ${codigo} - Corralon Progreso`;

  return `
<!-- corralon-share-preview-meta -->
<meta property="og:type" content="product">
<meta property="og:site_name" content="Corralon Progreso">
<meta property="og:title" content="${htmlAttrSharePreview(title)}">
<meta property="og:description" content="${htmlAttrSharePreview(description)}">
<meta property="og:url" content="${htmlAttrSharePreview(url.toString())}">
<meta property="og:image" content="${htmlAttrSharePreview(image)}">
<meta property="og:image:secure_url" content="${htmlAttrSharePreview(image)}">
<meta property="og:image:alt" content="${htmlAttrSharePreview(title)}">
<meta property="og:image:width" content="1200">
<meta property="og:image:height" content="1200">
<meta name="twitter:card" content="summary_large_image">
<meta name="twitter:title" content="${htmlAttrSharePreview(title)}">
<meta name="twitter:description" content="${htmlAttrSharePreview(description)}">
<meta name="twitter:image" content="${htmlAttrSharePreview(image)}">
`;
}

async function tryCorralonSharePreview(request) {
  const url = new URL(request.url);
  const path = url.pathname.replace(/\/+$/, '') || '/';
  const host = url.hostname.toLowerCase();
  const shouldInject =
    request.method === 'GET' &&
    (host === 'corralonprogreso.com' || host.endsWith('.corralonprogreso.com')) &&
    (path === '/' || path.toLowerCase() === '/index.html') &&
    url.searchParams.has('articulo') &&
    url.searchParams.has('previewImg');

  if (!shouldInject) return null;

  const response = await fetch(request);
  const contentType = response.headers.get('content-type') || '';
  if (!contentType.toLowerCase().includes('text/html')) return response;

  const meta = buildCorralonShareMeta(url);
  if (!meta) return response;

  const html = await response.text();
  const headers = new Headers(response.headers);
  headers.set('content-type', 'text/html; charset=utf-8');
  headers.set('cache-control', 'public, max-age=0, must-revalidate');
  headers.delete('content-length');

  const output = html.replace(/<\/head>/i, `${meta}</head>`);
  return new Response(output, {
    status: response.status,
    statusText: response.statusText,
    headers
  });
}

export default {
  async scheduled() {
    const token = await getFirebaseToken();
    await procesarRecordatorios(token);
  },

  async fetch(request) {
    const routeUrl = new URL(request.url);

    if (ONEPACE_VIDEO_PATHS.has(routeUrl.pathname)) {
      try {
        return await handleOnePaceVideo(request);
      } catch (e) {
        return videoError(e.message || 'Error en proxy de video', 500);
      }
    }

    if (ONEPACE_CATALOG_PATHS.has(routeUrl.pathname)) {
      try {
        return await handleOnePaceCatalog(request);
      } catch (e) {
        return json({ ok: false, error: e.message || 'Error en catalogo de One Pace' }, 500);
      }
    }

    if (request.method === 'OPTIONS') {
      return new Response(null, { headers: JSON_HEADERS });
    }

    const previewResponse = await tryCorralonSharePreview(request);
    if (previewResponse) return previewResponse;

    try {
      const url = routeUrl;

      if (url.pathname.startsWith('/admin/')) {
        const auth = await requireAdmin(request, url.pathname.replace('/admin/', '') || 'admin');
        if (!auth.ok) {
          return json({ ok: false, error: auth.error }, auth.status || 403);
        }

        const adminToken = await getFirebaseToken();

        if (request.method === 'POST' && url.pathname === '/admin/auth/verify') {
          await insertAdminAudit({
            admin_email: auth.adminEmail,
            target_uid: auth.adminUid,
            action: 'admin_verify',
            status: 'success',
            detail: 'Acceso admin validado'
          });
          return json({ ok: true, adminEmail: auth.adminEmail });
        }

        if (request.method === 'GET' && url.pathname === '/admin/pending') {
          const items = await listarPendientesAdmin();
          return json({ ok: true, items });
        }

        if (request.method === 'POST' && url.pathname === '/admin/pending/aprobar') {
          const body = await request.json();
          const result = await aprobarRegistroPendienteInterno({ idValue: body.id || '', adminEmail: auth.adminEmail });
          return json(result, result.status || 200);
        }

        if (request.method === 'POST' && url.pathname === '/admin/pending/rechazar') {
          const body = await request.json();
          const result = await rechazarRegistroPendienteInterno({ idValue: body.id || '', adminEmail: auth.adminEmail });
          return json(result, result.status || 200);
        }

        if (request.method === 'GET' && url.pathname === '/admin/users') {
          const users = await listarUsuariosFirebase(adminToken);
          return json({ ok: true, items: users });
        }

        if (request.method === 'POST' && url.pathname === '/admin/users/disable') {
          const body = await request.json();
          const uid = String(body.uid || '').trim();
          if (!uid) return json({ ok: false, error: 'Falta uid' }, 400);
          const user = await lookupFirebaseUser(adminToken, uid);
          if (!user) return json({ ok: false, error: 'Usuario no encontrado' }, 404);
          await setFirebaseUserDisabled(adminToken, uid, true);
          await insertAdminAudit({
            admin_email: auth.adminEmail,
            action: 'disable_user',
            target_email: user.email || '',
            target_uid: uid,
            status: 'success',
            detail: 'Usuario deshabilitado'
          });
          return json({ ok: true });
        }

        if (request.method === 'POST' && url.pathname === '/admin/users/enable') {
          const body = await request.json();
          const uid = String(body.uid || '').trim();
          if (!uid) return json({ ok: false, error: 'Falta uid' }, 400);
          const user = await lookupFirebaseUser(adminToken, uid);
          if (!user) return json({ ok: false, error: 'Usuario no encontrado' }, 404);
          await setFirebaseUserDisabled(adminToken, uid, false);
          await insertAdminAudit({
            admin_email: auth.adminEmail,
            action: 'enable_user',
            target_email: user.email || '',
            target_uid: uid,
            status: 'success',
            detail: 'Usuario habilitado'
          });
          return json({ ok: true });
        }

        if (request.method === 'POST' && url.pathname === '/admin/users/delete') {
          const body = await request.json();
          const uid = String(body.uid || '').trim();
          if (!uid) return json({ ok: false, error: 'Falta uid' }, 400);
          const user = await lookupFirebaseUser(adminToken, uid);
          if (!user) return json({ ok: false, error: 'Usuario no encontrado' }, 404);
          await deleteFirebaseUser(adminToken, uid);
          await insertAdminAudit({
            admin_email: auth.adminEmail,
            action: 'delete_user',
            target_email: user.email || '',
            target_uid: uid,
            status: 'success',
            detail: 'Usuario eliminado'
          });
          return json({ ok: true });
        }
                if (request.method === 'POST' && url.pathname === '/admin/users/login-as') {
          const body = await request.json();
          const uid = String(body.uid || '').trim();
          if (!uid) return json({ ok: false, error: 'Falta uid' }, 400);
          const user = await lookupFirebaseUser(adminToken, uid);
          if (!user) return json({ ok: false, error: 'Usuario no encontrado' }, 404);
          if (user.disabled) return json({ ok: false, error: 'El usuario está deshabilitado' }, 403);

          const customToken = await createFirebaseCustomToken(uid);

          await insertAdminAudit({
            admin_email: auth.adminEmail,
            action: 'login_as_user',
            target_email: user.email || '',
            target_uid: uid,
            status: 'success',
            detail: 'Admin inició sesión como usuario'
          });

          return json({ ok: true, customToken, email: user.email || '', uid });
        }

        if (request.method === 'GET' && url.pathname === '/admin/audit') {
          const items = await listarAuditAdmin();
          return json({ ok: true, items });
        }

        return json({ ok: false, error: 'Ruta admin no encontrada' }, 404);
      }

      if (request.method === 'GET' && url.pathname === '/registro-pendiente') {
        const result = await obtenerRegistroPendiente(url.searchParams.get('token') || '');
        return json(result, result.status || 200);
      }

      if (request.method === 'POST' && url.pathname === '/registro-pendiente') {
        const body = await request.json();
        const result = await crearRegistroPendiente(body);
        return json(result, result.status || 200);
      }

      if (request.method === 'POST' && url.pathname === '/registro-pendiente/aprobar') {
        const body = await request.json();
        const result = await aprobarRegistroPendiente(body.token || '');
        return json(result, result.status || 200);
      }

      if (request.method === 'POST' && url.pathname === '/registro-pendiente/rechazar') {
        const body = await request.json();
        const result = await rechazarRegistroPendiente(body.token || '');
        return json(result, result.status || 200);
      }

      if (request.method !== 'POST') {
        return json({ ok: false, error: 'Method not allowed' }, 405);
      }

      const body = await request.json();
      const token = await getFirebaseToken();

      if (body.tipo === 'gastoCompartido') {
        const result = await procesarGastoCompartido(token, body);
        return json(result);
      }

      if (body.tipo === 'solicitarReset') {
        const result = await solicitarReset(token, body.email);
        return json(result, result.ok ? 200 : 400);
      }

      if (body.tipo === 'confirmarReset') {
        const result = await confirmarReset(token, body.email, body.codigo, body.nuevaPass);
        return json(result, result.ok ? 200 : 400);
      }

      return json({ ok: false, error: 'Tipo no reconocido' }, 400);
    } catch (e) {
      return json({ ok: false, error: e.message || 'Error interno' }, 500);
    }
  }
};
