Public Function TotalCalculado2() As Double
    Dim precio As Double, Preciodto1 As Double, Preciodto2 As Double, Preciodto3 As Double
    Dim PorcFlete As Double, Multip As Double, Divivido As Double
    Dim iva As Double, res As Double

    ' Tomar valores de los controles
    precio = Nz(Me.Controls("Prelista"), 0)
    Preciodto1 = Nz(Me.Preciodto1, 0)
    Preciodto2 = Nz(Me.Preciodto2, 0)
    Preciodto3 = Nz(Me.Preciodto3, 0)
    PorcFlete = Nz(Me.PorcFlete, 0)
    iva = Nz(Me.iva, 0)
    Multip = Nz(Me.Multip, 0)
    Divivido = Nz(Me.Divivido, 1)   ' cuidado: si es 0, división por cero

    If iva > 1 Then iva = iva / 100   ' convierte 21 ? 0.21

    ' Subtotal sin IVA
    res = precio * (1 - Preciodto1) * (1 - Preciodto2) * (1 - Preciodto3) * (1 + PorcFlete) * Multip / Divivido

    ' Lógica según casilla verif_quitariva
    If Me.verif_quitariva = True Then
        ' Casilla marcada ? excluir IVA y dividir por quitariva
        If iva <> 0 Then
            res = res / (1 + iva)
        End If
    Else
        ' Casilla desmarcada ? incluir IVA normalmente
        res = res * (1 + iva)
    End If

    TotalCalculado2 = res
End Function
